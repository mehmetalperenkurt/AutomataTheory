        var stringPattern = "\".*\"";
        var arrayPattern = "\\[.*\\]";
        if (inputText.value != "") {

            var text = "";

            let patternStr = new RegExp(stringPattern,'g');
            let patternArr = new RegExp(arrayPattern,'g');

            let str = "<span class = foundStr>$&</span>";
            let arr = "<span class = foundArr>$&</span>";

            text = inputText.value.replace(patternArr, arr);
            text = text.replace(patternStr, str);
